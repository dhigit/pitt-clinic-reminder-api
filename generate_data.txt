import pymysql

def savePatient(patient_id, name):
    
    sql = """
        insert into tpatient (patient_id, patient_name) values ({patient_id}, '{name}')
    """.format(
        name = name,
        patient_id = patient_id
    )
    
    con = pymysql.connect(
        'localhost', 
        'root',
        '', 
        'reservation')

    with con:

        cur = con.cursor()
        cur.execute(sql)
